148  template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>> auto coefs2eval(T const &gc, double t) const {
149 
150  if (r != gc.shape(0)) throw std::runtime_error("First dim of g != DLR rank r.");
151 
152  // Scalar-valued Green's functions are handled differently than matrix-valued Green's functions
153 
154  if constexpr (T::rank == 1) {
155 
156  // TODO: can be further optimized to reduce # exponential evals.
157  // Evaluate DLR expansion
158  S g = 0;
159  if (t >= 0) {
160  for (int l = 0; l < r; ++l) { g += k_it_abs(t, dlr_rf(l)) * gc(l); }
161  } else {
162  for (int l = 0; l < r; ++l) { g += k_it_abs(-t, -dlr_rf(l)) * gc(l); }
163  }
164 
165  return g;
166  } else {
167 
168  // Reshape gc to matrix w/ first dimension r
169  auto gc_rs = nda::reshape(gc, r, gc.size() / r);
170 
171  // Get output shape
172  std::array<long, T::rank - 1> shape_out;
173  for (int i = 0; i < T::rank - 1; ++i) { shape_out[i] = gc.shape(i + 1); }
174 
175  // Get vector of evaluation of DLR expansion at a point
176  auto kvec = build_evalvec(t);
177 
178  // Evaluate DLR expansion, reshape to original dimensions (with first
179  // dimension summed out), and return
180  return nda::reshape(transpose(nda::matrix_const_view<S>(gc_rs)) * kvec, shape_out);
181  }

207  template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>>
208  typename T::regular_type fitvals2coefs(nda::vector_const_view<double> t, T const &g) const {
209 
210  // Check first dimension of g equal to length of t
211  int n = t.size();
212  if (n != g.shape(0)) throw std::runtime_error("First dim of g must be equal to length of t.");
213 
214  // Get matrix for least squares fitting: columns are DLR basis functions
215  // evaluating at data points t. Must built in Fortran layout for
216  // compatibility with LAPACK.
217  auto kmat = nda::matrix<S, F_layout>(n, r); // Make sure matrix has same scalar type as g
218  for (int j = 0; j < r; ++j) {
219  for (int i = 0; i < n; ++i) { kmat(i, j) = k_it(t(i), dlr_rf(j)); }
220  }
221 
222  // Reshape g to matrix w/ first dimension n, and put in Fortran layout for
223  // compatibility w/ LAPACK
224  auto g_rs = nda::matrix<S, F_layout>(nda::reshape(g, n, g.size() / n));
225 
226  // Solve least squares problem to obtain DLR coefficients
227  auto s = nda::vector<double>(r); // Singular values (not needed)
228  int rank = 0; // Rank of system matrix (not needed)
229  nda::lapack::gelss(kmat, g_rs, s, 0.0, rank);
230 
231  auto gc_shape = g.shape(); // Output shape is same as g...
232  gc_shape[0] = r; // ...with first dimension n replaced by r
233  auto gc = typename T::regular_type(gc_shape); // Output array: output type is same as g
234  reshape(gc, r, g.size() / n) = g_rs(nda::range(r), _); // Place result in output array
235 
236  return gc;
237  }

296  template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>>
297  typename T::regular_type convolve(double beta, statistic_t statistic, T const &fc, T const &gc, bool time_order = false) const {
298 
299  if (r != fc.shape(0) || r != gc.shape(0)) throw std::runtime_error("First dim of input arrays must be equal to DLR rank r.");
300  if (fc.shape() != gc.shape()) throw std::runtime_error("Input arrays must have the same shape.");
301 
302  // TODO: implement bosonic case and remove
303  if (statistic == 0) throw std::runtime_error("imtime_ops::convolve not yet implemented for bosonic Green's functions.");
304 
305  // Initialize convolution, if it hasn't been done already
306  if (!time_order & hilb.empty()) { convolve_init(); }
307  if (time_order & thilb.empty()) { tconvolve_init(); }
308 
309  // Get view of helper matrices based on time_order flag
310  auto hilb_v = (time_order ? nda::matrix_const_view<double>(thilb) : nda::matrix_const_view<double>(hilb));
311  auto tcf2it_v = (time_order ? nda::matrix_const_view<double>(ttcf2it) : nda::matrix_const_view<double>(tcf2it));
312 
313  if constexpr (T::rank == 1) { // Scalar-valued Green's function
314 
315  // Take array view of fc and gc
316  auto fca = nda::array_const_view<S, 1>(fc);
317  auto gca = nda::array_const_view<S, 1>(gc);
318 
319  // Diagonal contribution
320  auto h = arraymult(tcf2it_v, make_regular(fca * gca));
321 
322  // Off-diagonal contribution
323  auto tmp = fca * arraymult(hilb_v, gca) + gca * arraymult(hilb_v, fca);
324  return beta * (h + arraymult(cf2it, make_regular(tmp)));
325 
326  } else if (T::rank == 3) { // Matrix-valued Green's function
327 
328  // Diagonal contribution
329  auto fcgc = nda::array<S, 3>(fc.shape()); // Product of coefficients of f and g
330  for (int i = 0; i < r; ++i) { fcgc(i, _, _) = arraymult(fc(i, _, _), gc(i, _, _)); }
331  auto h = arraymult(tcf2it_v, fcgc);
332 
333  // Off-diagonal contribution
334  auto tmp1 = arraymult(hilb_v, fc);
335  auto tmp2 = arraymult(hilb_v, gc);
336  for (int i = 0; i < r; ++i) { tmp1(i, _, _) = arraymult(tmp1(i, _, _), gc(i, _, _)) + arraymult(fc(i, _, _), tmp2(i, _, _)); }
337 
338  return beta * (h + arraymult(cf2it, tmp1));
339 
340  } else {
341  throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
342  }
343  }

366  template <nda::MemoryMatrix Tf, nda::MemoryArray Tg, nda::Scalar Sf = nda::get_value_t<Tf>, nda::Scalar Sg = nda::get_value_t<Tg>,
367  nda::Scalar S = typename std::common_type<Sf, Sg>::type>
368  typename Tg::regular_type convolve(Tf const &fconv, Tg const &g) const {
369 
370  if (r != g.shape(0)) throw std::runtime_error("First dim of input g must be equal to DLR rank r.");
371 
372  if constexpr (Tg::rank == 1) { // Scalar-valued Green's function
373 
374  if (fconv.shape(1) != g.shape(0)) throw std::runtime_error("Input array dimensions incompatible.");
375 
376  return arraymult(fconv, g);
377 
378  } else if (Tg::rank == 3) { // Matrix-valued Green's function
379 
380  if (fconv.shape(1) != g.shape(0) * g.shape(1)) throw std::runtime_error("Input array dimensions incompatible.");
381 
382  int norb1 = g.shape(1);
383  int norb2 = g.shape(2);
384 
385  auto h = nda::array<S, 3>(r, norb1, norb2);
386  reshape(h, r * norb1, norb2) = arraymult(fconv, reshape(g, r * norb1, norb2));
387 
388  return h;
389 
390  } else {
391  throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
392  }
393  }

439  template <nda::MemoryArray T, nda::Scalar S = nda::get_value_t<T>>
440  nda::matrix<S> convmat(double beta, statistic_t statistic, T const &fc, bool time_order = false) const {
441 
442  if (r != fc.shape(0)) throw std::runtime_error("First dim of input array must be equal to DLR rank r.");
443 
444  // TODO: implement bosonic case and remove
445  if (statistic == 0) throw std::runtime_error("imtime_ops::convmat not yet implemented for bosonic Green's functions.");
446 
447  // Initialize convolution, if it hasn't been done already
448  if (!time_order & hilb.empty()) { convolve_init(); }
449  if (time_order & thilb.empty()) { tconvolve_init(); }
450 
451  // Get view of helper matrices based on time_order flag
452  auto hilb_v = (time_order ? nda::matrix_const_view<double>(thilb) : nda::matrix_const_view<double>(hilb));
453  auto tcf2it_v = (time_order ? nda::matrix_const_view<double>(ttcf2it) : nda::matrix_const_view<double>(tcf2it));
454 
455  if constexpr (T::rank == 1) { // Scalar-valued Green's function
456 
457  // First construct convolution matrix from DLR coefficients to DLR grid
458  // values
459  auto fconv = nda::matrix<S>(r, r); // Matrix of convolution by f
460 
461  // Diagonal contribution (given by diag(tau_k) * K(tau_k, om_l) * diag(fc_l))
462  for (int k = 0; k < r; ++k) {
463  for (int l = 0; l < r; ++l) { fconv(k, l) = tcf2it_v(k, l) * fc(l); }
464  }
465 
466  // Off-diagonal contribution (given by K * (diag(hilb*fc) +
467  // (diag(fc)*hilb)), where K is the matrix K(dlr_it(k), dlr_rf(l))
468  auto tmp1 = arraymult(hilb_v, fc); // hilb * fc
469  auto tmp2 = nda::matrix<S>(r, r);
470  for (int k = 0; k < r; ++k) {
471  for (int l = 0; l < r; ++l) {
472  tmp2(k, l) = fc(k) * hilb_v(k, l); // diag(fc) * hilb
473  }
474  tmp2(k, k) += tmp1(k); // diag(fc)*hilb + diag(hilb*fc)
475  }
476  fconv += arraymult(cf2it, tmp2);
477 
478  // Then precompose with DLR grid values to DLR coefficients matrix
479  if constexpr (nda::have_same_value_type_v<T, decltype(it2cf.lu)>) {
480  nda::lapack::getrs(transpose(it2cf.lu), fconv, it2cf.piv); // Note: lapack effectively tranposes fconv by fortran reordering here
481  } else {
482  // getrs requires matrix and rhs to have same value type
483  nda::lapack::getrs(transpose(nda::matrix<S>(it2cf.lu)), fconv, it2cf.piv);
484  }
485 
486  return beta * fconv;
487 
488  } else if (T::rank == 3) { // Matrix-valued Green's function
489 
490  int norb1 = fc.shape(1);
491  int norb2 = fc.shape(2);
492 
493  // First construct convolution matrix from DLR coefficients to DLR grid
494  // values
495  auto fconv = nda::matrix<S>(r * norb1, r * norb2); // Matrix of convolution by f
496  auto fconv_rs = nda::reshape(fconv, r, norb1, r, norb2); // Array view to index into fconv for conevenience
497 
498  // Diagonal contribution (given by diag(tau_k) * K(tau_k, om_l) * diag(fc_l))
499  for (int k = 0; k < r; ++k) {
500  for (int l = 0; l < r; ++l) { fconv_rs(k, _, l, _) = tcf2it_v(k, l) * fc(l, _, _); }
501  }
502 
503  // Off-diagonal contribution (given by K * (diag(hilb*fc) +
504  // (diag(fc)*hilb)), where K is the matrix K(dlr_it(k), dlr_rf(l))
505  auto tmp1 = arraymult(hilb_v, fc); // hilb * fc
506  auto tmp2 = nda::array<S, 4>(r, norb1, r, norb2);
507  for (int k = 0; k < r; ++k) {
508  for (int l = 0; l < r; ++l) {
509  tmp2(k, _, l, _) = fc(k, _, _) * hilb_v(k, l); // diag(fc) * hilb
510  }
511  tmp2(k, _, k, _) += tmp1(k, _, _); // diag(fc)*hilb + diag(hilb*fc)
512  }
513  fconv_rs += arraymult(cf2it, tmp2);
514 
515  // Then precompose with DLR grid values to DLR coefficients matrix
516 
517  // Transpose last two indices to put make column DLR index the last
518  // index
519  auto fconvtmp = nda::matrix<S>(r * norb1 * norb2, r);
520  auto fconvtmp_rs = nda::reshape(fconvtmp, r, norb1, norb2, r);
521  for (int i = 0; i < norb2; ++i) {
522  for (int k = 0; k < r; ++k) { fconvtmp_rs(_, _, i, k) = fconv_rs(_, _, k, i); }
523  }
524 
525  // Do the solve
526  if constexpr (nda::have_same_value_type_v<T, decltype(it2cf.lu)>) {
527  nda::lapack::getrs(transpose(it2cf.lu), fconvtmp, it2cf.piv); // Note: lapack effectively tranposes fconv by fortran reordering here
528  } else {
529  // getrs requires matrix and rhs to have same value type
530  nda::lapack::getrs(transpose(nda::matrix<S>(it2cf.lu)), fconvtmp, it2cf.piv);
531  }
532 
533  // Transpose back
534  for (int i = 0; i < norb2; ++i) {
535  for (int k = 0; k < r; ++k) { fconv_rs(_, _, k, i) = fconvtmp_rs(_, _, i, k); }
536  }
537 
538  return beta * fconv;
539 
540  } else {
541  throw std::runtime_error("Input arrays must be rank 1 (scalar-valued Green's function) or 3 (matrix-valued Green's function).");
542  }
543  }

599  void convolve_init() const {
600 
601  hilb = nda::matrix<double>(r, r);
602  tcf2it = nda::matrix<double>(r, r);
603 
604  // "Discrete Hilbert transform" matrix -(1-delta_jk)/(dlr_rf(j) -
605  // dlr_rf(k)), scaled by beta
606  for (int j = 0; j < r; ++j) {
607  for (int k = 0; k < r; ++k) {
608  if (j == k) {
609  hilb(j, k) = 0;
610  } else {
611  hilb(j, k) = 1.0 / (dlr_rf(j) - dlr_rf(k));
612  }
613  }
614  }
615 
616  // Matrix which applies DLR coefficients to imaginary time grid values
617  // transformation matrix, and then multiplies the result by tau, the
618  // imaginary time variable
619  for (int j = 0; j < r; ++j) {
620  for (int k = 0; k < r; ++k) {
621  if (dlr_it(j) > 0) {
622  tcf2it(j, k) = -(dlr_it(j) + k_it(1.0, dlr_rf(k))) * cf2it(j, k);
623  } else {
624  tcf2it(j, k) = -(dlr_it(j) - k_it(0.0, dlr_rf(k))) * cf2it(j, k);
625  }
626  }
627  }
628  }

639  void tconvolve_init() const {
640 
641  thilb = nda::matrix<double>(r, r);
642  ttcf2it = nda::matrix<double>(r, r);
643 
644  // "Discrete Hilbert transform" matrix
645  // -(1-delta_jk)*K(0,dlr_rf(k))/(dlr_rf(j) - dlr_rf(k)) for time-ordered
646  // convolution, scaled by beta
647  for (int j = 0; j < r; ++j) {
648  for (int k = 0; k < r; ++k) {
649  if (j == k) {
650  thilb(j, k) = 0;
651  } else {
652  thilb(j, k) = k_it(0.0, dlr_rf(k)) / (dlr_rf(k) - dlr_rf(j));
653  }
654  }
655  }
656 
657  // Matrix which applies K(0,dlr_rf(j)) multiplication, then DLR
658  // coefficients to imaginary time grid values transformation matrix, and
659  // then multiplies the result by tau, the imaginary time variable
660  for (int j = 0; j < r; ++j) {
661  for (int k = 0; k < r; ++k) {
662  if (dlr_it(j) > 0) {
663  ttcf2it(j, k) = dlr_it(j) * cf2it(j, k) * k_it(0.0, dlr_rf(k));
664  } else {
665  ttcf2it(j, k) = (1 + dlr_it(j)) * cf2it(j, k) * k_it(0.0, dlr_rf(k));
666  }
667  }
668  }
669  }

681 
682  refl = nda::matrix<double>(r, r);
683 
684  // Matrix of reflection acting on DLR coefficients and returning values at
685  // DLR nodes
